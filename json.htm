Json é uma variação do JavaScript Object, podendo ser representado de 3 formas

var json = { nome:'Sam', 'fone':9988 };
console.log(json.nome)    //---Sam
console.log(json['nome']) //---Sam
var json = [{ 'nome':'Sam', 'fone':9988 }, { 'nome':'Sam', 'fone':9977 }];
console.log(json[0].nome)        //---Sam
console.log(Array.isArray(json)) //---true
      
var json = { 'cliente': [
 {'nome':'Sam', 'fone':9988 },
 {'nome':'Ana', 'fone':9977 }
]}

console.log(Array.isArray(json.cliente)) //---true
console.log(json.cliente[0].nome)        //---Sam

Converter uma string para Json com parse(), situação que ocorre quando o Ajax recebe dados de um WebService

var texto = '{"nome":"Sam","fone":9988}';
var json = JSON.parse(texto);
console.log(json.fone) //---9988
